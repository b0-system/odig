type unop =
| Flip

flip p x y <=> p y x

| Trans

trans p == transitive closure

type binop =
| Union
| Inter
| Diff
| Join

join (x,y) (y,z) == (x,z)

| Product

product (x1,y1) (x2,y2) == (x1,y1,x2,y2)
