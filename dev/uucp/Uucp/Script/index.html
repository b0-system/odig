val compare : t -> t -> int

compare s s' is Pervasives.compare s s'.

val pp : Stdlib.Format.formatter -> t -> unit

pp ppf s prints an unspecified representation of s on ppf.

val script : Stdlib.Uchar.t -> t

script u is u's Script property.

val script_extensions : Stdlib.Uchar.t -> t list

script_extension u is u's Script_Extensions property. The list is never empty.
